package eu.diversify.trio.core.requirements;

/**
 * The three atomic logical operators
 */
public enum LogicalOperator {

    NOTHING(0),
    REQUIRE(1),
    NOT(2),
    OR(3),
    AND(4);
    
    public final int index;

    private LogicalOperator(int index) {
        this.index = index;
    }

}

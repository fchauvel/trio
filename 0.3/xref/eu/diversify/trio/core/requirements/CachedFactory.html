142     public Requirement createRequire(String componentName) {
143         final int hash = componentName.hashCode();
144         Requirement result = terminals.get(hash);
145         if (result == null) {
146             result = new Require(componentName);
147             terminals.put(hash, result);
148         }
149         return result;
150     }

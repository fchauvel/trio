35  package eu.diversify.trio.core.requirements.random;
36  
37  import eu.diversify.trio.core.requirements.Nothing;
38  import eu.diversify.trio.core.requirements.RequirementFactory;
39  import eu.diversify.trio.core.requirements.Requirement;
40  import java.util.List;
41  import java.util.Random;
42  
43  /**
44   * Generate a random sequence of commands, sent to the builder
45   */
46  public final class Goal {
47  
48      private int variableCount;
49      private final Random random;
50      private final Builder generation;
51  
52      public Goal(int variableCount) {
53          this(variableCount, new CachedLiteralFactory(variableCount));
54      }
55  
56      public Goal(int variableCount, RequirementFactory factory) {
57          this.variableCount = variableCount;
58          this.random = new Random();
59          this.generation = new Builder(factory);
60  
61      }
62  
63      public void setVariableCount(int variableCount) {
64          this.variableCount = variableCount;
65      }
66  
67      public Requirement build(int desiredSize) {
68          if (desiredSize == 0) {
69              return Nothing.getInstance();
70          }
71  
72          generation.reset(desiredSize);
73          do {
74              final Command command = pickAny(generation.possibleCommands());
75              command.randomize(this);
76              command.sendTo(generation);
77          } while (generation.isRunning());
78  
79          return generation.getResult();
80      }
81  
82      private Command pickAny(List<Command> candidates) {
83          if (candidates.isEmpty()) {
84              throw new IllegalArgumentException("Cannot pick from an empty collection");
85          }
86          if (candidates.size() == 1) {
87              return candidates.get(0);
88          }
89          return candidates.get(random.nextInt(candidates.size()));
90      }
91  
92      public int randomLeafIndex() {
93          return random.nextInt(variableCount);
94      }
95  
96  }

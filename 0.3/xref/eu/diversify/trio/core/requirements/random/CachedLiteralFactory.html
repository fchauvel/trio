18  package eu.diversify.trio.core.requirements.random;

20  import eu.diversify.trio.core.requirements.Conjunction;
21  import eu.diversify.trio.core.requirements.Disjunction;
22  import eu.diversify.trio.core.requirements.Negation;
23  import eu.diversify.trio.core.requirements.Require;
24  import eu.diversify.trio.core.requirements.Requirement;
25  import eu.diversify.trio.core.requirements.RequirementFactory;

27  /**
28   * Pre-build the createRequire objects for given number of components, that will be
29   shared among requirements.
30   */
31  public class CachedLiteralFactory extends RequirementFactory {
32  
33      private static final int CAPACITY = 10191;
34      private final int size;
35      private final Require[] prebuilt;
36  
37      public CachedLiteralFactory(int limit) {
38          assert limit < CAPACITY:
39                  "The given limit " + limit + " exceeds the actual capacity " + CAPACITY;
40          
41          this.size = limit;
42          this.prebuilt = new Require[CAPACITY];
43          for (int i = 0; i < size; i++) {
44              prebuilt[i] = new Require("C" + i);
45          }
46      }
47      
48      @Override
49      public Requirement createRequire(int hash) {
50          return (prebuilt[hash] != null) ? prebuilt[hash] : new Require("C" + hash);
51      }
52  
53      @Override
54      public Requirement createRequire(String componentName) {
55          final int hashCode = componentName.hashCode();
56          assert hashCode >= 0: "calculating remainder of a negative hashcode";
57          int index = hashCode % CAPACITY;
58          return (prebuilt[index] != null) ? prebuilt[index] : new Require(componentName);
59      }
60  
61      @Override
62      public Requirement createConjunction(Requirement left, Requirement right) {
63          return new Conjunction(left, right);
64      }
65  
66      @Override
67      public Requirement createDisjunction(Requirement left, Requirement right) {
68          return new Disjunction(left, right);
69      }
70  
71      @Override
72      public Requirement createNegation(Requirement operand) {
73          return new Negation(operand);
74      }
75  
76  }

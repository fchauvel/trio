18  package eu.diversify.trio.core.requirements.stats;
19  
20  import eu.diversify.trio.core.DefaultSystemVisitor;
21  import eu.diversify.trio.core.requirements.*;
22  import java.util.Arrays;
23  
24  /**
25   * Count logical operators in a given requirement
26   */
27  public class OperatorDistribution extends DefaultSystemVisitor {
28  
29      private final int[] counts;
30  
31      public OperatorDistribution() {
32          counts = new int[LogicalOperator.values().length];
33      }
34  
35      public double disjunctionRatio() {
36          return ((double) counts[LogicalOperator.OR.index]) / operatorCount();
37      }
38  
39      public double conjunctionRatio() {
40          return ((double) counts[LogicalOperator.AND.index]) / operatorCount();
41      }
42  
43      public double negationRatio() {
44          return ((double) counts[LogicalOperator.NOT.index]) / operatorCount();
45      }
46  
47      public int operatorCount() {
48          int total = 0;
49          for (LogicalOperator operator: LogicalOperator.values()) {
50              total += counts[operator.index];
51          }
52          return total;
53      }
54  
55      @Override
56      public void exit(Nothing nothing) {
57          counts[LogicalOperator.NOTHING.index]++;
58      }
59  
60      @Override
61      public void exit(Require require) {
62          counts[LogicalOperator.REQUIRE.index]++;
63      }
64  
65      @Override
66      public void exit(Negation negation) {
67          counts[LogicalOperator.NOT.index]++;
68      }
69  
70      @Override
71      public void exit(Disjunction disjunction) {
72          counts[LogicalOperator.OR.index]++;
73      }
74  
75      @Override
76      public void exit(Conjunction conjunction) {
77          counts[LogicalOperator.AND.index]++;
78      }
79  
80      public int[] counts() {
81          return Arrays.copyOf(counts, counts.length);
82      }
83  
84  }

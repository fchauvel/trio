35  package eu.diversify.trio.core.requirements;
36  
37  import eu.diversify.trio.core.SystemPart;
38  import eu.diversify.trio.simulation.Topology;
39  
40  /**
41   * Logical expression representing the components whose activity (or inactivity)
42   * is required.
43   */
44  public abstract class Requirement implements SystemPart {
45  
46      public final Requirement and(Requirement right) {
47          return new Conjunction(this, right);
48      }
49  
50      public final Requirement or(Requirement right) {
51          return new Disjunction(this, right);
52      }
53  
54      public final Requirement not() {
55          return new Negation(this);
56      }
57  
58      public final Requirement xor(Requirement right) {
59          return or(right).and(right.not());
60      }
61       
62      public abstract boolean isSatisfiedBy(Topology topology);
63     
64      @Override
65      public abstract String toString();
66  
67  }

29  /**
30   * Dispatch visit events (i.e., enter, exit) to other visitors, to factor out
31   * the cost of traversing large models.
32   */
33  public class Dispatcher extends DefaultSystemVisitor {
34  
35      private final List<SystemVisitor> listeners;
36  
37      public Dispatcher(SystemVisitor... listeners) {
38          this(Arrays.asList(listeners));
39      }
40  
41      public Dispatcher(List<SystemVisitor> listeners) {
42          eu.diversify.trio.util.Require.notNull(listeners, "Invalid 'null' value provided as listeners!");
43          eu.diversify.trio.util.Require.notEmpty(listeners, "Invalid empty list [] provided as listeners!");
44          this.listeners = new ArrayList<SystemVisitor>(listeners);
45      }
46  
47      @Override
48      public final void enter(Nothing nothing) {
49          for (SystemVisitor each: listeners) {
50              each.enter(nothing);
51          }
52      }
53

55      public final void exit(Nothing nothing) {
56          for (SystemVisitor each: listeners) {
57              each.exit(nothing);
58          }
59      }

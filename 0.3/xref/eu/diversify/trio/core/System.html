35  /**
36   * Represent the specification of a system under study. A collection component,
37   * uniquely identified by their name.
38   */
39  public class System implements SystemPart {
40  
41      public static final String DEFAULT_NAME = "Anonymous";
42  
43      private final String name;
44      private final List<Component> components;
45      private final Map<String, Integer> indexByName;
46      private final Map<String, Tag> tags;
47  
48      public System(Component... components) {
49          this(DEFAULT_NAME, Arrays.asList(components), new ArrayList<Tag>());
50      }
51  
52      public System(Collection<Component> components) {
53          this(DEFAULT_NAME, components, new ArrayList<Tag>());
54      }
55  
56      public System(Collection<Component> components, Collection<Tag> tags) {
57          this(DEFAULT_NAME, components, tags);
58      }
59  
60      public System(String name, Collection<Component> components, Collection<Tag> tags) {
61          Require.notNull(name, "Invalid value 'null', given as system name!");
62          this.name = name;
63  
64          Require.notNull(components, "Invalid value 'null' given as list of components!");
65          Require.notEmpty(components, "Invalid value [] given as list of components!");
66          this.components = new ArrayList<Component>(components);
67          this.indexByName = new HashMap<String, Integer>(components.size());
68          int counter = 0;
69          for (Component each: components) {
70              this.indexByName.put(each.getName(), counter);
71              counter++;
72          }
73  
74          this.tags = new HashMap<String, Tag>();
75          for(Tag eachTag: tags) {
76              this.tags.put(eachTag.getLabel(), eachTag);
77          }
78      }
79  
80      private String validate(String componentName) throws IllegalArgumentException {
81          if (!indexByName.containsKey(componentName)) {
82              final String error = String.format("Syntax error: Unknown component '%s' (Components are: %s)", componentName, indexByName.keySet());
83              throw new IllegalArgumentException(error);
84          }
85          return componentName;
86      }
87  
88      public String getName() {
89          return name;
90      }
91  
92      /**
93       * @return the number of components in the systems
94       */
95      public int size() {
96          return this.components.size();
97      }

99      /**
100      * @param name the name of the component whose index is needed
101      * @return the index of the given component, is it exists, or null otherwise
102      */
103     public int indexOf(String name) {
104         return indexByName.get(validate(name));
105     }
106 
107     /**
108      * @param componentName the name of component of interest
109      * @return true if there exists a component with the given name, false
110      * otherwise.
111      */
112     public boolean hasComponentNamed(String componentName) {
113         return indexByName.containsKey(componentName);
114     }
115 
116     public Collection<SystemPart> subParts() {
117         final List<SystemPart> subparts = new ArrayList<SystemPart>(components.size() + tags.size());
118         subparts.addAll(components);
119         subparts.addAll(tags.values());
120         return subparts;
121     }

131     public Set<String> taggedAs(String tag) {

133     }

135     @Override
136     public int hashCode() {
137         int hash = 7;
138         hash = 41 * hash + this.name.hashCode();
139         hash = 41 * hash + this.components.hashCode();
140         hash = 41 * hash + this.tags.hashCode();
141         return hash;
142     }

144     @Override
145     public boolean equals(Object obj) {
146         if (obj == null) {
147             return false;
148         }
149         if (getClass() != obj.getClass()) {
150             return false;
151         }
152         final System other = (System) obj;
153         return other.name.equals(name) && other.components.equals(components) && other.tags.equals(tags);
154     }

156     public Set<String> getComponentNames() {

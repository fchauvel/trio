18  package eu.diversify.trio.filter;
19  
20  import eu.diversify.trio.core.System;
21  import java.util.HashSet;
22  import java.util.Set;
23  
24  /**
25   * Logical disjunction
26   */
27  public class Or extends Filter {
28  
29      private final Filter left;
30      private final Filter right;
31  
32      public Or(Filter left, Filter right) {
33          this.left = left;
34          this.right = right;
35      }
36  
37      @Override
38      public Set<String> resolve(System system) {
39          final Set<String> results = new HashSet<String>();
40          results.addAll(left.resolve(system));
41          results.addAll(right.resolve(system));
42          return results;
43      }
44  
45      @Override
46      public String toString() {
47          return String.format("(%s or %s)", left.toString(), right.toString());
48      }
49  
50      
51      
52  }

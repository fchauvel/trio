66      private Command(String[] commandLine) {
67          final List<String> parameters = Arrays.asList(commandLine);
68          outputFile = DEFAULT_OUTPUT_FILE;
69          runCount = DEFAULT_RUN_COUNT;
70          observation = DEFAULT_OBSERVATION;
71          control = DEFAULT_CONTROL;
72          extractOptions(parameters);
73      }

75      private void extractOptions(List<String> parameters) {
76          final Iterator<String> command = parameters.iterator();
77          while (command.hasNext()) {
78              final String parameter = command.next();
79              if (parameter.matches(".*\\.trio")) {
80                  inputFile = parameter;
81                  break;
82  
83              } else if (parameter.startsWith("--")) {
84                  parseLongOptions(parameter);
85  
86              } else if (parameter.startsWith("-")) {
87                  parseShortOptions(command, parameter);
88  
89              } else {
90                  final String error = String.format("Unknown argument '%s'. Expecting either a path to a TRIO file or an option.", parameter);
91                  throw new IllegalArgumentException(error);
92              }
93          }
94          if (inputFile == null) {
95              throw new IllegalArgumentException("Missing input TRIO file!");
96          }
97      }
98  
99      private void parseShortOptions(final Iterator<String> command, final String eachParameter) throws IllegalArgumentException {
100         if (!command.hasNext()) {
101             final String error = String.format("Missing value after '%s'", eachParameter);
102             throw new IllegalArgumentException(error);
103         }
104         if (eachParameter.equals("-o")) {
105             observation = command.next();
106 
107         } else if (eachParameter.equals("-c")) {
108             control = command.next();
109 
110         } else if (eachParameter.equals("-t")) {
111             outputFile = command.next();
112 
113         } else if (eachParameter.equals("-r")) {
114             extractRunCountFrom(command.next());
115 
116         } else {
117             final String error = String.format("Unknown short option '%s'", eachParameter);
118             throw new IllegalArgumentException(error);
119         }
120     }

122     private void parseLongOptions(final String eachParameter) throws IllegalArgumentException {
123         final String parts[] = eachParameter.split("=");
124         if (parts[0].equals("--trace")) {
125             outputFile = parts[1];
126 
127         } else if (parts[0].equals("--observe")) {
128             observation = parts[1];
129 
130         } else if (parts[0].equals("--control")) {
131             control = parts[1];
132 
133         } else if (parts[0].equals("--runs")) {
134             extractRunCountFrom(parts[1]);
135 
136         } else {
137             final String error = String.format("Unknown long option '%s'! Expecting either '--runs' or '--output'", parts[0]);
138             throw new IllegalArgumentException(error);
139         }
140     }

142     private void extractRunCountFrom(String text) throws IllegalArgumentException {
143         try {
144             runCount = Integer.parseInt(text);
145             if (runCount < 1) {
146                 final String error = String.format("Run count shall be at least 1! (found %d)", runCount);
147                 throw new IllegalArgumentException(error);
148             }
149 
150         } catch (NumberFormatException ex) {
151             final String error = String.format("Invalid number of runs '%s", text);
152             throw new IllegalArgumentException(error);
153 
154         }
155     }

157     public void execute(Trio trio, OutputStream output) throws IOException {
158         final Configuration config = Configuration.forProduction();
159 
160         PrintStream out = null;
161 
162         try {
163             out = new PrintStream(output, AUTO_FLUSH, UTF_8);
164 
165             out.println(config.version() + " -- " + config.description());
166             out.println(config.copyright());
167             out.println("Licensed under " + config.license());
168             out.println();
169 
170             final System system = trio.loadSystemFrom(inputFile);
171             
172             trio.validate(system);
173             
174             out.println("SYSTEM: " + system.getName());
175             final RandomFailureSequence scenario = new RandomFailureSequence(system, observation(), control());
176             out.println("SCENARIO: " + this.format(scenario));
177 
178             final DataSet data = trio.run(scenario, runCount);
179             trio.saveDataAs(data, outputFile);
180 
181             report(trio.analyse(data), out);
182 
183             out.println();
184             out.println("That's all folks!");
185             
186         } catch (FileNotFoundException ex) {
187             out.println("Error: Unable to open '" + inputFile + "'");
188             
189         } catch (IOException ex) {
190             throw new RuntimeException("Unable to write on the given output stream", ex);
191 
192         } catch (InvalidSystemException ex) {
193             out.println("Invalid model:");
194             for(Inconsistency each: ex.getErrors()) {
195                 out.println(" - " + each.getDescription());
196             }
197                     
198         } catch (SyntaxError ex) {
199             out.println("Invalid model:");
200             for(String eachError: ex.getErrors()) {
201                 out.println(" - " + eachError);
202             }
203         
204         } finally {
205             if (out != null) {
206                 out.close();
207             }
208         }
209     }

221     private Filter parseTag(String tag) {

223             return All.getInstance();
224         }

228     private void report(Analysis analyzer, PrintStream out) {
229         out.println("INDICATORS:");
230         out.printf(" + Robustness: %.4f%n", getRobustness(analyzer).distribution().mean());
231         out.println(" + Five most sensitive components:");
232         int counter = 1;
233         for (Map.Entry<String, Double> entry: getLoss(analyzer).distribution().byDecreasingExpectedValue(Distribution.mean).entrySet()) {
234             out.printf("   %d: %.4e %s%n", counter++, entry.getValue(), entry.getKey().replaceAll("inactivate", ""));
235             if (counter > 5) {
236                 break;
237             }
238         }
239         reportHarmfulSequences(out, analyzer);
240     }

242     public void reportHarmfulSequences(PrintStream out, Analysis analyzer) {
243         out.println(" + Five most harmful sequences:");
244         int counter = 1;
245         for (Map.Entry<String, Double> entry: getFragility(analyzer).distribution().byDecreasingExpectedValue(Distribution.mean).entrySet()) {
246             out.printf("   %d: %.4e %s%n", counter++, entry.getValue(), entry.getKey().replaceAll("inactivate", ""));
247             if (counter > 5) {
248                 break;
249             }
250         }
251     }

265     private String format(Scenario scenario) {
266         String observation = String.format("'%s' layer", scenario.getObservation().toString());
267         if (observation.contains("*")) {
268             observation = "system";
269         }
270         String control = String.format("'%s' layer", scenario.getControl().toString());
271         if (control.contains("*")) {
272             control = "system";
273         }
274         return String.format("Robustness of the %s to failure in the %s", observation, control);
275     }

277     public static String usage() {

279         return "Usage: trio [options] input.trio" + EOL
280                 + "where 'options' are:" + EOL
281                 + "  -o, --observe=TAG  the tag of the components whose activity shall be observed" + EOL
282                 + "  -c, --control=TAG  the tag of the components whose activity shall be controlled" + EOL
283                 + "  -r, --runs=INTEGER the number of sample for statistical evidence" + EOL
284                 + "  -t, --trace=FILE   the file where the generated data shall be stored" + EOL
285                 + "Example: trio -o result.csv --run=10000 system.trio";
286     }

20  package eu.diversify.trio.data;
21  
22  import eu.diversify.trio.codecs.DataFormat;
23  import eu.diversify.trio.simulation.Action;
24  import java.util.ArrayList;
25  import java.util.List;
26  
27  import static eu.diversify.trio.simulation.actions.AbstractAction.*;
28  
29  /**
30   * The data collected by the recorder
31   */
32  public class Trace {
33  
34      private final int observationCapacity;
35      private final int controlCapacity;
36      private final List<State> states;
37  
38      public Trace(int capacity) {
39          this(capacity, capacity);
40      }
41  
42      public Trace(int observationCapacity, int controlCapacity) {
43          this.observationCapacity = observationCapacity;
44          this.controlCapacity = controlCapacity;
45          this.states = new ArrayList<State>();
46          states.add(new State(none(), 0, observationCapacity, controlCapacity, 0));
47      }
48      
49      public String label() {
50          StringBuilder buffer = new StringBuilder();
51          buffer.append('[');
52          for(int i=0 ; i<states.size() ;i++) {
53              buffer.append(states.get(i).getTrigger().toString());
54              if (i < states.size()-1) {
55                  buffer.append(", ");
56              }
57          }
58          buffer.append(']');
59          return buffer.toString();
60      }
61  
62      public int getObservationCapacity() {
63          return observationCapacity;
64      }
65  
66      public int getControlCapacity() {
67          return controlCapacity;
68      }
69      
70      public void accept(DataSetListener listener) {
71          listener.enterTrace(this);
72          for (State eachState: states) {
73              eachState.accept(listener);
74          }
75          listener.exitTrace(this);
76      }
77  
78      public List<Integer> disruptionLevels() {
79          final List<Integer> levels = new ArrayList<Integer>();
80          for (State eachState: states) {
81              levels.add(eachState.getDisruptionLevel());
82          }
83          return levels;
84      }
85  
86      public void record(Action action, int activityLevel) {
87          record(action, activityLevel, activityLevel);
88      }
89  
90      public void record(Action action, int activeAndObserved, int activeAndControlled) {
91          final State next = this.current().update(action, activeAndObserved, activeAndControlled);
92          this.states.add(next);
93      }
94  
95      private State current() {
96          if (states.isEmpty()) {
97              throw new IllegalStateException("A trace shall have at least one state!");
98          }
99          final int lastIndex = states.size() - 1;
100         return states.get(lastIndex);
101     }
102 
103     public State afterDisruption(int level) {
104         for (State eachState: states) {
105             if (eachState.getDisruptionLevel() == level) {
106                 return eachState;
107             }
108         }
109         return null;
110     }
111 
112     public int length() {
113         return states.size() - 1;
114     }
115 
116     public String to(DataFormat format, int index) {
117         final String EOL = java.lang.System.lineSeparator();
118         final StringBuilder builder = new StringBuilder();
119         for (State eachState: states) {
120             builder.append(eachState.to(format, index)).append(EOL);
121         }
122         return builder.toString();
123     }

125 }

18  package eu.diversify.trio.data;
19  
20  import java.io.OutputStream;
21  import java.io.PrintStream;
22  import java.io.UnsupportedEncodingException;
23  
24  /**
25   * Format a data set into a CSV file
26   */
27  public class CSVFormatter extends AbstractDataSetListener {
28  
29      private final PrintStream out;
30      private int traceCounter;
31  
32      public CSVFormatter(OutputStream output) {
33          try {
34              out = new PrintStream(output, true, "UTF-8");
35          
36          } catch (UnsupportedEncodingException ex) {
37              throw new RuntimeException("Unable to write UTF-8", ex);
38          
39          } 
40      }
41  
42      @Override
43      protected void byDefault() throws UnsupportedOperationException {
44          // Do nothing
45      }
46  
47      @Override
48      public void enterDataSet(DataSet dataSet) {
49          traceCounter = 0;
50          out.print("sequence" + FIELD_SEPARATOR);
51          out.print("action" + FIELD_SEPARATOR);
52          out.print("disruption" + FIELD_SEPARATOR);
53          out.print("activity" + FIELD_SEPARATOR);
54          out.println("loss");
55      }
56  
57      @Override
58      public void enterTrace(Trace trace) {
59          traceCounter++;
60      }
61  
62      @Override
63      public void enterState(State state) {
64          out.print(traceCounter + FIELD_SEPARATOR);
65          out.print(state.getTrigger().toString() + FIELD_SEPARATOR);
66          out.print(state.getDisruptionLevel() + FIELD_SEPARATOR);
67          out.print(state.getObservedActivityLevel() + FIELD_SEPARATOR);
68          out.println(state.getLoss());
69      }
70  
71      private static final String FIELD_SEPARATOR = ",";
72  
73  }

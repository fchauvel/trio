33  /**
34   * Facade to the various 'builder' object that build sub part of the system
35   */
36  public class Builder {
37  
38      /**
39       * Factory method that increase the readibility of the calls
40       */
41      public static Builder build() {
42          return new Builder();
43      }
44  
45      public Requirement requirementFrom(String text) {
46          TrioLexer lexer = new TrioLexer(new ANTLRInputStream(text));
47          CommonTokenStream tokens = new CommonTokenStream(lexer);
48          TrioParser parser = new TrioParser(tokens);
49          ParseTree tree = parser.requirements();
50          return tree.accept(new ExpressionBuilder());
51      }
52  
53      public Component componentFrom(String text) {
54          TrioLexer lexer = new TrioLexer(new ANTLRInputStream(text));
55          CommonTokenStream tokens = new CommonTokenStream(lexer);
56          TrioParser parser = new TrioParser(tokens);
57          ParseTree tree = parser.component();
58          return tree.accept(new ComponentBuilder());
59      }
60  
61      public System systemFrom(String text) {
62          TrioLexer lexer = new TrioLexer(new ANTLRInputStream(text));
63          CommonTokenStream tokens = new CommonTokenStream(lexer);
64          TrioParser parser = new TrioParser(tokens);
65          ParseTree tree = parser.system();
66          return tree.accept(new SystemBuilder());
67      }
68  
69      public System systemFrom(InputStream input) throws IOException, SyntaxError {
70          SystemErrorListener listener = new SystemErrorListener();
71          TrioLexer lexer = new TrioLexer(new ANTLRInputStream(input));
72          lexer.removeErrorListeners();
73          lexer.addErrorListener(listener);
74          CommonTokenStream tokens = new CommonTokenStream(lexer);
75          TrioParser parser = new TrioParser(tokens);
76          parser.removeErrorListeners();
77          parser.addErrorListener(listener);
78          ParseTree tree = parser.system();
79          listener.abortIfAnyError();
80          return tree.accept(new SystemBuilder());
81      }

83  }

96      @Test
97      public void scalability() throws FileNotFoundException {
98          final PrintStream log = new PrintStream(new FileOutputStream("target/scalability.csv"));
99          log.println("duration,size,density,disjunction,conjunction,negation");
100 
101         final Generator generate = new Generator();
102 
103         final Random random = new Random();
104 
105         final int MAX_COMPONENT_COUNT = 1000;
106         final int MIN_COMPONENT_COUNT = 1;
107         final int runCount = 500; 
108 
109         for (int i = 0; i < runCount; i++) {
110             int size = MIN_COMPONENT_COUNT + random.nextInt(MAX_COMPONENT_COUNT - MIN_COMPONENT_COUNT);
111             final Distribution meanValenceDistribution = Distribution.uniform(0, size);
112             final double mean = meanValenceDistribution.sample();
113             final Distribution density = Distribution.normal(mean, mean / 4);
114             final System system = generate.system(size, density);
115             final Scenario scenario = new RandomFailureSequence(system);
116             double duration = durationOf(scenario);
117 
118             final Statistics stats = new Statistics();
119             evaluate(stats).on(system);
120             
121             log.print(duration);
122             log.print(",");
123             log.print(system.size());
124             log.print(",");
125             log.print(stats.toCsvENtry());
126         }
127 
128         log.close();
129     }

131     public double averageDurationOf(Scenario scenario, int runCount) {
132         long total = 0L;
133         for (int i = 0; i < runCount; i++) {
134             total += durationOf(scenario);
135         }
136         return total * (1D / runCount);
137     }
138 
139     public long durationOf(Scenario scenario) {
140         long start = java.lang.System.currentTimeMillis();
141         trio.run(scenario, 1);
142         long end = java.lang.System.currentTimeMillis();
143         long duration = end - start;
144         return duration;
145     }

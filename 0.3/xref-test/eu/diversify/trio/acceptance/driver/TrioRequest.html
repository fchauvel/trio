19  package eu.diversify.trio.acceptance.driver;
20  
21  import eu.diversify.trio.Configuration;
22  import java.io.IOException;
23  import java.nio.file.Files;
24  import java.nio.file.Paths;
25  
26  /**
27   * Request the execution of Trio
28   */
29  public class TrioRequest {
30      
31      private final Configuration config;
32      private String pathToTopology;
33      private String destination;
34      private String observation;
35      private String control;
36          
37      public TrioRequest(String path, String trace, String observation, String control) {
38          this.config = Configuration.forTest();
39          this.pathToTopology = path;
40          this.destination = trace;
41          this.observation = observation;
42          this.control = control;
43      }
44      public String getPathToTopology() {
45          return (pathToTopology == null) ? "" : pathToTopology;
46      }
47      
48      public String getDestination() {
49          return (destination == null)? "" : destination;
50      }    
51  
52      public String getObservation() {
53          return observation;
54      }
55  
56      public String getControl() {
57          return control;
58      }
59      
60      
61      public TrioResponse execute() throws IOException, InterruptedException {
62          String command = String.format("java -jar %s -o %s -c %s -t %s %s", config.jarFile(), observation, control, destination, pathToTopology);
63          RunAsThread run = new RunAsThread(config.installationDirectory(), command.split("\\s+"));
64          return new TrioResponse(run);
65      } 
66  
67      public void deleteGeneratedFiles() throws IOException {
68          Files.delete(Paths.get(config.installationDirectory()));
69      }
70  
71  }

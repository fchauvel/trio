19  package eu.diversify.trio.acceptance.driver;
20  
21  import junit.framework.TestCase;
22  import org.junit.Test;
23  import org.junit.runner.RunWith;
24  import org.junit.runners.JUnit4;
25  
26  /**
27   * Specification of the Trio response object
28   */
29  @RunWith(JUnit4.class)
30  public class TrioResponseTest extends TestCase {
31  
32      /**
33       * A dummy run whose stdout and stderr are simply given as parameters;
34       */
35      private static class DummyRun implements Run {
36  
37          private final String stdout;
38          private final String stderr;
39  
40          public DummyRun(String stdout) {
41              this(stdout, "");
42          }
43  
44          public DummyRun(String stdout, String stderr) {
45              this.stdout = stdout;
46              this.stderr = stderr;
47          }
48  
49          public String getStandardError() {
50              return stderr;
51          }
52  
53          public String getStandardOutput() {
54              return stdout;
55          }
56  
57      }
58  
59      @Test
60      public void assertRobustnessAboutShouldDetectRobustness() {
61          final TrioResponse sut = new TrioResponse(new DummyRun(" + Robustness: 0.1234"));
62          sut.assertRobustnessAbout(0.1234, 1e-6);
63      }
64  
65      @Test(expected = AssertionError.class)
66      public void assertRobustnessAboutShouldDetectMissingRobustness() {
67          final TrioResponse sut = new TrioResponse(new DummyRun("blablabla bla\n\nblablabla"));
68          sut.assertRobustnessAbout(0.1234, 1e-6);
69      }
70  
71      @Test
72      public void assertRobustnessAboutShouldDetectWithinMultilineText() {
73          final String stdout
74                  = "blablabla bla\n"
75                  + "blablabla\n"
76                  + "Robustness: 0.1234\n"
77                  + "blablabla blabla";
78          final TrioResponse sut = new TrioResponse(new DummyRun(stdout));
79          sut.assertRobustnessAbout(0.1234, 1e-6);
80      }
81  
82      @Test
83      public void shouldDetectInvalidModels() {
84          final TrioResponse sut = new TrioResponse(new DummyRun(" + Invalid model: \"Unknown keyword 'foo'\""));
85          sut.assertInvalidSystemDetected();
86      }
87  
88      @Test(expected = AssertionError.class)
89      public void shouldNotTakeOtherErrorsForInvalidModels() {
90          final TrioResponse sut = new TrioResponse(new DummyRun(" +  Error: \"Unknown keyword 'foo'\""));
91          sut.assertInvalidSystemDetected();
92      }
93  
94  }

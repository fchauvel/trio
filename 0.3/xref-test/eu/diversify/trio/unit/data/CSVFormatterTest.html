39  @RunWith(JUnit4.class)
40  public class CSVFormatterTest extends TestCase {
41  
42      private static final String EOL = System.lineSeparator();
43      private final CSVFormatter formatter;
44      private final ByteArrayOutputStream csv;
45  
46      public CSVFormatterTest() {
47          csv = new ByteArrayOutputStream();
48          formatter = new CSVFormatter(csv);
49      }
50  
51      @Test
52      public void stateShouldBeProperlyFormatted() {
53          final State state = new State(inactivate("X"), 1, 5, 3, 1);
54  
55          state.accept(formatter);
56  
57          final String expectation = "0,inactivate X,1,5,1" + EOL;
58  
59          assertThat(csv.toString(), is(equalTo(expectation)));
60      }
61  
62      @Test
63      public void traceShouldBeProperlyFormatted() {
64          final Trace trace = new Trace(10, 5);
65          trace.record(inactivate("A"), 9, 4);
66          trace.record(inactivate("B"), 4, 3);
67          trace.record(inactivate("C"), 3, 2);
68          trace.record(inactivate("D"), 0, 1);
69  
70          trace.accept(formatter);
71  
72          final String expectation
73                  = "1,none,0,10,0" + EOL
74                  + "1,inactivate A,1,9,1" + EOL
75                  + "1,inactivate B,2,4,5" + EOL
76                  + "1,inactivate C,3,3,1" + EOL
77                  + "1,inactivate D,4,0,3" + EOL;
78  
79          assertThat(csv.toString(), is(equalTo(expectation)));
80      }
81  
82      @Test
83      public void shouldFormatDataSetCorrectly() {
84          final DataSet dataset = new DataSet();
85          final Trace trace = new Trace(10, 5);
86          trace.record(inactivate("A"), 9, 4);
87          trace.record(inactivate("B"), 4, 3);
88          trace.record(inactivate("C"), 3, 2);
89          trace.record(inactivate("D"), 0, 1);
90          dataset.include(trace);
91          dataset.include(trace);
92  
93          dataset.accept(formatter);
94  
95          final String expectation
96                  = "sequence,action,disruption,activity,loss" + EOL
97                  + "1,none,0,10,0" + EOL
98                  + "1,inactivate A,1,9,1" + EOL
99                  + "1,inactivate B,2,4,5" + EOL
100                 + "1,inactivate C,3,3,1" + EOL
101                 + "1,inactivate D,4,0,3" + EOL
102                 + "2,none,0,10,0" + EOL
103                 + "2,inactivate A,1,9,1" + EOL
104                 + "2,inactivate B,2,4,5" + EOL
105                 + "2,inactivate C,3,3,1" + EOL
106                 + "2,inactivate D,4,0,3" + EOL;
107 
108         assertThat(csv.toString(), is(equalTo(expectation)));
109     }

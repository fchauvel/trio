44  @RunWith(Parameterized.class)
45  public class TrioTest extends TestCase {
46  
47      private final String name;
48      private final Scenario scenario;
49      private final Analysis analysis;
50      private final String metric;
51      private final double expectedValue;
52  
53      public TrioTest(String name, Scenario scenario, Analysis analysis, String metric, double expectedValue) {
54          this.name = name;
55          this.scenario = scenario;
56          this.analysis = analysis;
57          this.metric = metric;
58          this.expectedValue = expectedValue;
59      }
60  
61      @Test
62      public void test() {
63          final Trio sut = new Trio();
64          DataSet trace = sut.run(scenario, 1000);
65          trace.accept(analysis);
66          final double actual = analysis.metric(metric).distribution().mean();
67          assertThat(actual, is(closeTo(expectedValue, 1e-1)));
68      }

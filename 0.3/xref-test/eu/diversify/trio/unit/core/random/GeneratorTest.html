34          final int RUN_COUNT = 10;
35          final int COMPONENT_COUNT = 1000;
36  
37          final Generator random = new Generator();
38  
39          System.out.println("Generation of " + RUN_COUNT + " system(s) of " + COMPONENT_COUNT + " component(s).");
40        
41          long totalDuration = 0;
42          for (int index = 0; index < RUN_COUNT; index++) {
43                    
44              final long start = System.currentTimeMillis();
45              random.system(COMPONENT_COUNT, Distribution.uniform(0, COMPONENT_COUNT));
46              final long end = System.currentTimeMillis();
47  
48              final long duration = end - start;
49        
50              totalDuration += duration;
51          }
52  
53          double averageInSeconds = ((double) totalDuration) / 10 / 1000 ;
54          System.out.println();
55          System.out.println("Average duration: " + averageInSeconds + " s.");
56      }

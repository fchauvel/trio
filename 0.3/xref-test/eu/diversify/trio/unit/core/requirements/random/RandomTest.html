59          final int RUN_COUNT = 10000;
60          final int SIZE_LIMIT = 100;
61          final int VARIABLE_CAPACITY = 10000;
62  
63          final Goal goal = new Goal(VARIABLE_CAPACITY, new CachedLiteralFactory(10000));
64  
65          final CachedFactory factory = new CachedFactory();
66          final Requirement[] results = new Requirement[RUN_COUNT];
67  
68          long start = System.currentTimeMillis();
69          for (int runIndex = 0; runIndex < RUN_COUNT; runIndex++) {
70              int size = new Random().nextInt(SIZE_LIMIT);
71              results[runIndex] = goal.build(size);
72          }
73          long end = System.currentTimeMillis();
74  
75          System.out.println(RUN_COUNT + " random requirement built in " + (end - start) + " ms.");
76      }
